function drawBicycle(x, y, angle, color, name, isCrashed, isFrozen, memeTxt) { ctx.save(); ctx.scale(cameraZoom, cameraZoom); ctx.translate((x - camera.x), (y - camera.y)); // 依據世界相機定位 // 繪製迷因對話框 (不受旋轉影響) if (isFrozen && memeTxt) { ctx.save(); ctx.translate(0, -170); // 往上移避免擋到名字和車子 ctx.font = '900 48px sans-serif'; // 字體大三倍 (16px -> 48px) let tw = ctx.measureText(memeTxt).width; let bw = tw + 60; let bh = 80; ctx.fillStyle = '#ffffff'; ctx.strokeStyle = '#131722'; ctx.lineWidth = 4; ctx.beginPath(); ctx.roundRect(-bw/2, -bh/2, bw, bh, 15); ctx.fill(); ctx.stroke(); // 泡泡 ctx.beginPath(); ctx.moveTo(-15, bh/2); ctx.lineTo(0, bh/2 + 25); ctx.lineTo(15, bh/2); ctx.fill(); ctx.stroke(); // 尾巴 ctx.fillStyle = '#f23645'; ctx.textAlign = 'center'; ctx.textBaseline = 'middle'; ctx.fillText(memeTxt, 0, 0); ctx.restore(); } ctx.rotate(angle); // 名字 ID (往上移，避免跟頭重疊) ctx.fillStyle = color; ctx.font = 'bold 24px sans-serif'; ctx.textAlign = 'center'; ctx.fillText(name, 0, -90); // 從 -60 改為 -90 // 車身 function updatePhysics(dt) { if (bike.finished) return; let dtMultiplier = dt * 60; // 迷因發作：突然煞車並大喊 (保證發作) if (bike.memeTriggerXs.length > 0 && bike.x > bike.memeTriggerXs[0]) { bike.memeTriggerXs.shift(); // 移除已經發作過的點 bike.isMemeFrozen = true; bike.memeTimer = 4.5; // 定格時間增加三倍 (1.5 -> 4.5 秒) bike.memeText = memeQuotes[Math.floor(Math.random() * memeQuotes.length)]; bike.vx = 0; bike.vy = 0; // 真實人聲語音播報 (TTS) if ('speechSynthesis' in window) { let msg = new SpeechSynthesisUtterance(bike.memeText); msg.lang = 'en-US'; // 英文發音 msg.rate = 1.1; // 稍微快一點比較激動 msg.pitch = Math.random() * 0.6 + 0.8; // 隨機音調增加趣味 window.speechSynthesis.speak(msg); } } if (bike.isMemeFrozen) { bike.memeTimer -= dt; if (!bike.isAirborne && bike.isGasBtnDown) { let accel = 0.66 * dtMultiplier; // 車速(加速度)變快 20% bike.vx += accel * Math.cos(trackInfo.angle); bike.vy += accel * Math.sin(trackInfo.angle); } bike.x += bike.vx * dtMultiplier; bike.y += bike.vy * dtMultiplier; let speedAlongTrack = bike.vx * Math.cos(newTrackInfo.angle) + bike.vy * Math.sin(newTrackInfo.angle); speedAlongTrack *= Math.pow(0.99, dtMultiplier); if (!bike.isGasBtnDown) speedAlongTrack *= Math.pow(0.95, dtMultiplier); // 最高速限限制加快 20% (18 -> 21.6) if (speedAlongTrack > 21.6) speedAlongTrack = 21.6; if (speedAlongTrack < -3.6) speedAlongTrack = -3.6; bike.vx = speedAlongTrack * Math.cos(newTrackInfo.angle); function gameLoop(timestamp) { if (!lastTime) lastTime = timestamp; let dt = (timestamp - lastTime) / 1000; lastTime = timestamp; if (dt > 0.1) dt = 0.1; updatePhysics(dt); drawTradingViewChart(); for (let id in players) { if (id !== myId) { const p = players[id]; // 讓其他連線對手的迷因也能發出人聲 if (p.memeFrozen && p.memeTxt && !p.hasSpoken) { if ('speechSynthesis' in window) { let msg = new SpeechSynthesisUtterance(p.memeTxt); msg.lang = 'en-US'; msg.rate = 1.1; window.speechSynthesis.speak(msg); } p.hasSpoken = true; } else if (!p.memeFrozen) { p.hasSpoken = false; // 沒發作時重置 } drawBicycle(p.x, p.y, p.rot, p.color, p.name, p.crashed, p.memeFrozen, p.memeTxt); } } drawBicycle(bike.x, bike.y, bike.angle, myColor, myName, bike.isCrashed, bike.isMemeFrozen, bike.memeText); syncNetwork(); checkFinish();
